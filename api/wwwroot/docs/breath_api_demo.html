BREATH API – R Vignette

Author: GeoModelLab Date: 2026

Overview

BREATH (Biophysical Rhythm of Ecosystem Activity & Health) simulates hourly GPP, RECO, and NEE for deciduous broadleaf forest pixels using:

  • VPRM (Vegetation Photosynthesis and Respiration Model) for carbon fluxes
  • SWELL phenology for seasonal leaf dynamics
  • NASA POWER weather data (hourly or daily disaggregated)

This vignette demonstrates the complete R workflow:

  1. Submit a simulation via the REST API
  2. Poll until the run completes
  3. Parse the hourly CSV results
  4. Compute annual carbon budgets and seasonal diagnostics
  5. Visualise fluxes and SWELL phenology

1 · Submit a simulation

A pixel is identified by "lat_lon" (4 decimal places). We simulate a deciduous broadleaf forest in the northern Apennines (Italy).

r submit
pixel    <- "44.1234_10.7654"
start_yr <- 2022L
end_yr   <- 2024L

payload <- list(
  settings = list(
    pixelsRun           = list(pixel),
    startYear           = start_yr,
    endYear             = end_yr,
    inputWeather        = "hourly",        # "hourly" | "daily" | "era5land"
    modelVariant        = "Circadian",     # "Baseline" | "Pheno" | "Circadian"
    calibration         = FALSE,           # TRUE = calibrate against MODIS EVI
    calibrationVariable = "Phenology",
    simplexes           = 3L,
    iterations          = 200L,
    parametersDataFile  = "photothermalRequirements.csv"
  )
)

resp <- request(paste0(BASE_URL, "/api/breath/run")) |>
  req_method("POST") |>
  req_body_json(payload) |>
  req_timeout(30) |>
  req_perform()

resp_body_json(resp)

2 · Poll until complete

r poll
wait_for_completion <- function(base_url, poll_interval = 10, timeout = 600) {
  deadline <- Sys.time() + timeout
  repeat {
    status <- request(paste0(base_url, "/api/breath/status")) |>
      req_timeout(10) |>
      req_perform() |>
      resp_body_json()

    cat(sprintf("\r  Status: %-12s", status$Status))

    if (status$Status == "Completed") {
      cat(sprintf("\nDone — duration: %s ms\n", status$DurationMs %||% "?"))
      return(invisible(status))
    }
    if (status$Status == "Failed")
      stop("Simulation failed. Check /api/breath/stream/logs for details.")
    if (Sys.time() > deadline)
      stop(sprintf("Simulation did not finish within %d s.", timeout))

    Sys.sleep(poll_interval)
  }
}

# Null-coalescing helper (base R < 4.4 compatible)
`%||%` <- function(a, b) if (!is.null(a)) a else b

wait_for_completion(BASE_URL)

3 · Download and parse results

GET /api/results/latest returns a CSV with one row per simulated hour.

r download
csv_raw <- request(paste0(BASE_URL, "/api/results/latest")) |>
  req_timeout(60) |>
  req_perform() |>
  resp_body_string()

df <- read_csv(csv_raw, show_col_types = FALSE) |>
  mutate(date = as.Date(date),
         year  = year(date),
         month = month(date),
         doy   = yday(date))

4 · Annual carbon budget

Daily mean fluxes (µmol CO₂ m⁻² s⁻¹) × 86 400 s day⁻¹ × 12.01 g mol⁻¹ × 10⁻⁶ mol µmol⁻¹ = gC m⁻² day⁻¹.

r annual-budget
UMOL_TO_GC_DAY <- 86400 * 12.01e-6

daily <- df |>
  group_by(date, year) |>
  summarise(across(c(GPP, RECO, NEE), mean, na.rm = TRUE), .groups = "drop") |>
  mutate(across(c(GPP, RECO, NEE), ~ .x * UMOL_TO_GC_DAY))

annual <- daily |>
  group_by(year) |>
  summarise(
    GPP_gC  = round(sum(GPP,  na.rm = TRUE), 1),
    RECO_gC = round(sum(RECO, na.rm = TRUE), 1),
    NEE_gC  = round(sum(NEE,  na.rm = TRUE), 1),
    CUE     = round(GPP_gC / RECO_gC, 3),
    Balance = if_else(NEE_gC < 0, "sink", "source"),
    .groups = "drop"
  )

knitr::kable(annual, caption = "Annual carbon budget (gC m⁻² yr⁻¹)")

6 · Seasonal diagnostics

r seasonal
seasons <- df |>
  mutate(season = case_when(
    month %in% c(12, 1, 2) ~ "DJF",
    month %in% 3:5          ~ "MAM",
    month %in% 6:8          ~ "JJA",
    TRUE                    ~ "SON"
  )) |>
  group_by(year, season) |>
  summarise(
    T_mean   = round(mean(t,    na.rm = TRUE), 1),
    P_total  = round(sum(p,     na.rm = TRUE) * 24, 0),  # mm/h → mm/day, summed
    GPP_mean = round(mean(GPP,  na.rm = TRUE), 3),
    NEE_mean = round(mean(NEE,  na.rm = TRUE), 3),
    .groups = "drop"
  ) |>
  mutate(season = factor(season, levels = c("DJF","MAM","JJA","SON"))) |>
  arrange(year, season)

7 · Phenological dates

When the Circadian or Pheno model variant is used, the output includes a phenophase column that can be used to extract phenological dates (DOY).

r phenology
pheno_col <- grep("phenophase", names(df), ignore.case = TRUE, value = TRUE)[1]

if (!is.na(pheno_col)) {
  pheno <- df |>
    rename(phase = all_of(pheno_col)) |>
    arrange(date) |>
    group_by(year) |>
    mutate(prev_phase = lag(phase, default = "")) |>
    filter(phase != prev_phase) |>
    summarise(
      SGS = doy[grepl("Growth",    phase, ignore.case = TRUE)][1],
      MAT = doy[grepl("Greendown", phase, ignore.case = TRUE)][1],
      SEN = doy[grepl("Senesc",    phase, ignore.case = TRUE)][1],
      EGS = doy[grepl("Dorm|Induct", phase, ignore.case = TRUE)][1],
      .groups = "drop"
    ) |>
    mutate(GSL = EGS - SGS)

  knitr::kable(pheno, caption = "Phenological dates (DOY)")
} else {
  cat("No phenophase column in output — run with modelVariant='Pheno' or 'Circadian'.\n")
}

9 · Parameter overrides

Override any internal model parameter by passing a parameterOverrides map. Keys follow the className_propertyName convention.

r overrides (eval=FALSE)
tuned_payload <- list(
  settings = list(
    pixelsRun           = list(pixel),
    startYear           = 2023L,
    endYear             = 2023L,
    inputWeather        = "hourly",
    modelVariant        = "Circadian",
    calibration         = FALSE,
    parametersDataFile  = "photothermalRequirements.csv",
    simplexes           = 3L,
    iterations          = 200L,
    calibrationVariable = "Phenology"
  ),
  parameterOverrides = list(
    parPhotosynthesis_maximumQuantumYieldOver = 0.08,  # α overstory (default 0.06)
    parPhotosynthesis_halfSaturationTree      = 350,   # k½ PAR (default 400)
    parRespiration_referenceRespiration       = 1.2    # R base (default 1.5)
  )
)

request(paste0(BASE_URL, "/api/breath/run")) |>
  req_method("POST") |>
  req_body_json(tuned_payload) |>
  req_perform()

wait_for_completion(BASE_URL)

10 · Custom weather data

By default BREATH downloads meteorological inputs automatically from NASA POWER. If you have your own station data, reanalysis output, or ERA5 downloads you can upload a daily weather CSV and the next simulation for that pixel will use it instead — skipping the remote download entirely.

10.1 · Required CSV format

The file must be comma-separated with the following columns (dot as decimal separator):

ColumnUnitNotes
Dateyyyy-MM-ddOne row per day
PARMJ m⁻² d⁻¹Photosynthetically active radiation
airTemperatureMaximum°CDaily maximum
airTemperatureMinimum°CDaily minimum
solarRadiationMJ m⁻² d⁻¹Global shortwave
dewPointTemperature°CUsed to compute VPD
precipitationmm d⁻¹Total daily rainfall
latitude°Pixel latitude (constant per file)

Retrieve a minimal template with:

# Fetch template
template <- request(paste0(BASE_URL, "/api/weather/template")) |>
  req_perform() |>
  resp_body_string()
cat(template)

10.2 · Upload

pixel_custom <- "44.6813_11.0217"

# Build a synthetic daily data frame (replace with real observations)
weather_df <- tibble::tibble(
  Date                    = seq(as.Date("2022-01-01"), as.Date("2024-12-31"), by = "day"),
  PAR                     = 5.0,
  airTemperatureMaximum   = 12.0,
  airTemperatureMinimum   =  2.0,
  solarRadiation          = 10.0,
  dewPointTemperature     = -1.0,
  precipitation           =  1.5,
  latitude                = 44.6813
)

weather_csv <- readr::format_csv(weather_df)

# Upload — stored server-side as {pixel}_weather.csv
resp_upload <- request(paste0(BASE_URL, "/api/weather/upload")) |>
  req_url_query(pixel = pixel_custom) |>
  req_method("POST") |>
  req_headers("Content-Type" = "text/csv") |>
  req_body_raw(weather_csv, type = "text/csv") |>
  req_perform()

resp_upload |> resp_body_json()

10.3 · Run with custom weather

After uploading, run the simulation as usual. BREATH detects the cached file and skips the NASA POWER call automatically.

custom_payload <- list(
